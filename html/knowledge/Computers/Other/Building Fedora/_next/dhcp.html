DHCP Installation

Purpose Login as root Install dhcp 
yum -y install dhcp

vi /etc/dhcp/dhcpd.conf

dG
i

#
#   DHCP Server Config File
#

ddns-update-style none;
ignore client-updates;

lease-file-name "/var/lib/dhcpd/dhcpd.leases";
authoritative;

option domain-name                      "local.austin.rr.com";
default-lease-time                      86400;   # 24 hours
max-lease-time                          172800;  # 48 hours

subnet 192.168.10.0 netmask 255.255.255.0 {
        option routers                  192.168.10.1;
        option subnet-mask              255.255.255.0;
        option broadcast-address        192.168.10.255;
        option domain-name-servers      $_CURRENT_IP;
        range   192.168.20.100 192.168.20.104;
}
Note this step
touch /install/dhcp_installation.$INSTALL_VERSION
Exit root 
exit
