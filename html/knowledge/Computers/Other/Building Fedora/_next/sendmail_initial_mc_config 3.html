Initial sendmail.mc configuration

Change to the mail directory 
cd /etc/mail

mkdir -p /Saved/etc/mail
Backup the original .mc file 
mkdir -p /Saved/etc/mail

cp sendmail.mc /Saved/etc/mail/sendmail.mc
Open sendmail.mc 
vi /etc/mail/sendmail.mc
Remove current contents and setup for insertion 
dG
i
Paste new file in place 
include(`/usr/share/sendmail-cf/m4/cf.m4')dnl
OSTYPE(`linux')dnl

m4 sendmail.mc > sendmail.cf
Send a test mail 
su - testuser

echo -e "Subject: This is the subject\nThis is the body" | /usr/lib/sendmail -F mickey.mouse@disney.com -t testuser
Check for the test mail 
mail

exit
Look at the logs 
cat /var/log/maillog

Note this step

/usr/local/bin/note.sh sendmail_initial_mc
