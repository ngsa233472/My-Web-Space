Create the default .bashrc 
vi /etc/skel/.bashrc
dG
i
# .bashrc

# Source global definitions
if [ -f /etc/bashrc ]; then
        . /etc/bashrc
fi

# User specific aliases and functions

alias c='clear'
alias l='ls -l'
alias bdf='df -hT'
alias h='history'
alias p='ps -ef'
alias vi='vim'

umask 077
ulimit -c 0
set -o emacs
set -b

case $- in *i*)
        if [ -e "/usr/bin/fortune" ]
        then
            /usr/bin/fortune
        fi
esac
Create the default .vimrc 
vi /etc/skel/.vimrc
colo desert
