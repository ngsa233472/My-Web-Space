\begin{align} 1.\ \int sin\ u\ du = -cos u \\ \\ 2.\ \int u^n\ du = {{u^{n+1}} \over {n+1}} \\ \\ 3.\ \int {1 \over u}\ du =\ ln \vert u \vert \\ \\ 4.\ \int sec^2\ u\ du = tan \ u\\ \\ 5.\ \int e^u \ du = e^u\ {1 \over ln}\ {du \over du}\\ \\ 6.\ \int \ sec\ u\ tan\ u\ du = sec\ u\\ \\ 7.\ \int \ a\ du = au \\ \\ 8.\ \int be^{-au} \ du = -{be^{-au} \over a} \\ \\ 9.\ \int {1 \over {\sqrt {a^2 + u^2}}}\ du = {1 \over a}sin^{-1}\ \big( {u \over a}\big)\\ \\ 10.\ \int tan \ u \ du = ln \vert \sec u \vert \\ \\ 11.\ \int \csc^2 \ u \ du = - \cot \ u\\ \\ 12.\ \int \sec \ u \ du = ln \vert \sec \ u + \tan \ u \vert\\ \\ 13.\ \int cot \ u \ du = ln \vert \sin \ u \vert\\ \\ 15.\ \int \tan^2 \ u \ du = \tan u \ - \ u \\ \\ 16.\ \int a^u \ du = {a^u \over ln \ a}\\ \\ 17.\ \int \csc \ u \cot \ u \ du = - \csc u\\ \\ 18.\ \int ln \ u \ du = u \ ln \ u - u\\ \\ 19.\ \int {1 \over {a^2 + u^2}}\ du = {1 \over a}tan^{-1}\ \big( {u \over a}\big)\\ integration by parts\\ \\ 20.\ \int u \ dv = uv \ \int v \ du\\ subsitution\\ \\ 21.\ \int {x \over {x^2 + 1}} dx \\ \\ u = x^2 + 1 \ and \ du = 2x \ dx\\ \\ \int {du \over 2u} = {1 \over 2} ln \vert u \vert = {1 \over 2} ln \vert x^2 + 1 \vert\\ \end{align}
